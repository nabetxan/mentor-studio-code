# Phase 2.3: i18n (日英切替) + Actions UI改善 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Japanese/English language switching to the webview UI, and improve the Actions buttons by replacing the copy icon with a mortar-board icon.

**Architecture:** Locale stored in `.mentor-studio.json` as a `locale` field (`"ja"` | `"en"`). A simple translations object maps keys to strings per locale. Locale flows from extension → webview via existing config message. Settings tab gets a language toggle. Actions buttons replace `codicon-copy` with `codicon-mortar-board` and keep `codicon-check` on copied state.

**Tech Stack:** TypeScript, React, plain CSS, VSCode Extension API, Vitest

```typescript
import type { Locale } from "@mentor-studio/shared";
import { useEffect, useRef, useState } from "react";
import { t } from "../i18n";
import { postMessage } from "../vscodeApi";

interface Snippet {
  id: string;
  titleKey: "actions.startNextTask" | "actions.reviewImplementation" | "actions.startReview" | "actions.startCheck";
  promptKey: "actions.prompt.startNextTask" | "actions.prompt.reviewImplementation" | "actions.prompt.startReview" | "actions.prompt.startCheck";
}

const SNIPPETS: Snippet[] = [
  { id: "start-next-task", titleKey: "actions.startNextTask", promptKey: "actions.prompt.startNextTask" },
  { id: "review-implementation", titleKey: "actions.reviewImplementation", promptKey: "actions.prompt.reviewImplementation" },
  { id: "start-review", titleKey: "actions.startReview", promptKey: "actions.prompt.startReview" },
  { id: "start-check", titleKey: "actions.startCheck", promptKey: "actions.prompt.startCheck" },
];

interface ActionsProps {
  locale: Locale;
}

export function Actions({ locale }: ActionsProps) {
  const [copiedId, setCopiedId] = useState<string | null>(null);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    return () => {
      if (timerRef.current !== null) {
        clearTimeout(timerRef.current);
      }
    };
  }, []);

  const handleCopy = (snippet: Snippet) => {
    if (timerRef.current !== null) {
      clearTimeout(timerRef.current);
    }
    setCopiedId(snippet.id);
    postMessage({ type: "copy", text: t(snippet.promptKey, locale) });
    timerRef.current = setTimeout(() => setCopiedId(null), 2000);
  };

  return (
    <div className="actions">
      <h3>{t("actions.title", locale)}</h3>
      <p className="actions-description">
        {t("actions.description", locale)}
      </p>
      <div className="snippet-list">
        {SNIPPETS.map((snippet) => (
          <button
            className={`snippet-btn${copiedId === snippet.id ? " copied" : ""}`}
            key={snippet.id}
            onClick={() => handleCopy(snippet)}
          >
            <span className="snippet-title">{t(snippet.titleKey, locale)}</span>
            <span className="snippet-icon">
              {copiedId === snippet.id ? (
                <>
                  <i className="codicon codicon-check" />
                  <span className="snippet-copied-text">{t("actions.copied", locale)}</span>
                </>
              ) : (
                <i className="codicon codicon-mortar-board" />
              )}
            </span>
          </button>
        ))}
      </div>
    </div>
  );
}
```

```typescript
private async updateLocale(locale: Locale): Promise<void> {
  const wsRoot = vscode.workspace.workspaceFolders?.[0]?.uri;
  if (!wsRoot) {
    return;
  }
  const configUri = vscode.Uri.joinPath(wsRoot, ".mentor-studio.json");
  try {
    const raw = await vscode.workspace.fs.readFile(configUri);
    const config = JSON.parse(
      Buffer.from(raw).toString(),
    ) as MentorStudioConfig;
    config.locale = locale;
    await vscode.workspace.fs.writeFile(
      configUri,
      Buffer.from(JSON.stringify(config, null, 2) + "\n"),
    );
    this.latestConfig = config;
    this.postMessage({ type: "config", data: config });
  } catch (err) {
    console.error("Failed to update locale", err);
  }
}
```
